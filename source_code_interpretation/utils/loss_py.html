    """Áî®Êù•‰ª£ÊõøFocalLoss
    QFocalLoss Êù•Ëá™General Focal LossËÆ∫Êñá: https://arxiv.org/abs/2006.04388
    Wraps Quality focal loss around existing loss_fcn(), 
    i.e. criteria = FocalLoss(nn.BCEWithLogitsLoss(), gamma=1.5)
    """
    # Wraps Quality focal loss around existing loss_fcn(), i.e. criteria = FocalLoss(nn.BCEWithLogitsLoss(), gamma=1.5)
    def __init__(self, loss_fcn, gamma=1.5, alpha=0.25):
        super().__init__()
        self.loss_fcn = loss_fcn  # must be nn.BCEWithLogitsLoss()
        self.gamma = gamma
        self.alpha = alpha
        self.reduction = loss_fcn.reduction
        self.loss_fcn.reduction = "none"  # required to apply FL to each element

    def forward(self, pred, true):
        loss = self.loss_fcn(pred, true)

        pred_prob = flow.sigmoid(pred)  # prob from logits
        alpha_factor = true * self.alpha + (1 - true) * (1 - self.alpha)
        modulating_factor = flow.abs(true - pred_prob) ** self.gamma
        loss = loss * (alpha_factor * modulating_factor)

        if self.reduction == "mean":
            return loss.mean()
        elif self.reduction == "sum":
            return loss.sum()
        else:  # 'none'
            return loss

                # m: [126] bool Â¶ÇÊûúÊòØTrueË°®Á§∫ÂΩìÂâçtarget‰∏≠ÂøÉÁÇπÊâÄÂú®ÁöÑÊ†ºÂ≠êÁöÑ‰∏ãËæπÊ†ºÂ≠ê‰πüÂØπËØ•targetËøõË°åÂõûÂΩí(ÂêéÁª≠ËøõË°åËÆ°ÁÆóÊçüÂ§±)

                # t: [126, 7] -> Â§çÂà∂5‰ªΩtarget[5, 126, 7]  ÂàÜÂà´ÂØπÂ∫îÂΩìÂâçÊ†ºÂ≠êÂíåÂ∑¶‰∏äÂè≥‰∏ãÊ†ºÂ≠ê5‰∏™Ê†ºÂ≠ê

                 # flow.zeros_like(gxy)[None]: [1, 126, 2]   off[:, None]: [5, 1, 2]  => [5, 126, 2]
                # jÁ≠õÈÄâÂêé: [378, 2]  ÂæóÂà∞ÊâÄÊúâÁ≠õÈÄâÂêéÁöÑÁΩëÊ†ºÁöÑ‰∏≠ÂøÉÁõ∏ÂØπ‰∫éËøô‰∏™Ë¶ÅÈ¢ÑÊµãÁöÑÁúüÂÆûÊ°ÜÊâÄÂú®ÁΩëÊ†ºËæπÁïåÔºàÂ∑¶Âè≥‰∏ä‰∏ãËæπÊ°ÜÔºâÁöÑÂÅèÁßªÈáè
                offsets = (flow.zeros_like(gxy)[None] + off[:, None])[j]
            else:
                t = targets[0]

            # Define

            # a, (b, c) = a.long().view(-1), bc.long().T  # anchors, image, class
            a, (b, c) = (
                a.contiguous().long().view(-1),
                bc.contiguous().long().T,
            )  # anchors, image, class

            # gij = (gxy - offsets).long()
            # È¢ÑÊµãÁúüÂÆûÊ°ÜÁöÑÁΩëÊ†ºÊâÄÂú®ÁöÑÂ∑¶‰∏äËßíÂùêÊ†á(ÊúâÂ∑¶‰∏äÂè≥‰∏ãÁöÑÁΩëÊ†º)  
            gij = (gxy - offsets).contiguous().long() 

            gi, gj = gij.T  # grid indices

            # Append

            # indices.append((b, a, gj.clamp_(0, shape[2] - 1), gi.clamp_(0, shape[3] - 1)))  # image, anchor, grid
            gi = gi.clamp(0, shape[3] - 1)
            gj = gj.clamp(0, shape[2] - 1)
            # b: image index  a: anchor index  gj: ÁΩëÊ†ºÁöÑÂ∑¶‰∏äËßíyÂùêÊ†á  gi: ÁΩëÊ†ºÁöÑÂ∑¶‰∏äËßíxÂùêÊ†á

            # tbix: xywh ÂÖ∂‰∏≠xy‰∏∫Ëøô‰∏™targetÂØπÂΩìÂâçgrid_cellÂ∑¶‰∏äËßíÁöÑÂÅèÁßªÈáè

            anch.append(anchors[a])  # anchors   ÂØπÂ∫îÁöÑÊâÄÊúâanchors
            tcls.append(c)  # class

        """

        lcls = flow.zeros(1, device=self.device)  # class loss
        lbox = flow.zeros(1, device=self.device)  # box loss
        lobj = flow.zeros(1, device=self.device)  # object loss
        # Ëé∑ÂæóÊ†áÁ≠æÂàÜÁ±ª,ËæπÊ°Ü,Á¥¢ÂºïÔºåanchors
        # ÊØè‰∏Ä‰∏™ÈÉΩÊòØappendÁöÑ Êúâfeature map‰∏™ 
        # ÈÉΩÊòØÂΩìÂâçËøô‰∏™feature map‰∏≠3‰∏™anchorÁ≠õÈÄâÂá∫ÁöÑÊâÄÊúâÁöÑtarget(3‰∏™grid_cellËøõË°åÈ¢ÑÊµã)

                # Áî®Ëøô‰∏™È¢ÑÊµãÂÄº‰∏éÊàë‰ª¨Á≠õÈÄâÁöÑËøô‰∏™grid_cellÁöÑÁúüÂÆûÊ°ÜËøõË°åÈ¢ÑÊµã(ËÆ°ÁÆóÊçüÂ§±)

                if self.nc > 1:  # cls loss (only if multiple classes)
                    # targets ÂéüÊú¨Ë¥üÊ†∑Êú¨ÊòØ0  ËøôÈáå‰ΩøÁî®smooth label Â∞±ÊòØcn
                    t = flow.full_like(pcls, self.cn, device=self.device)  # targets

            obji = self.BCEobj(pi[..., 4], tobj)

            # ‰∏ÄËà¨Êù•ËØ¥ÔºåÊ£ÄÊµãÂ∞èÁâ©‰ΩìÁöÑÈöæÂ∫¶Â§ß‰∏ÄÁÇπÔºåÊâÄ‰ª•‰ºöÂ¢ûÂä†Â§ßÁâπÂæÅÂõæÁöÑÊçüÂ§±Á≥ªÊï∞ÔºåËÆ©Ê®°ÂûãÊõ¥Âä†‰æßÈáçÂ∞èÁâ©‰ΩìÁöÑÊ£ÄÊµã
            lobj = lobj + (obji * self.balance[i])  # obj loss

                # Ëá™Âä®Êõ¥Êñ∞ÂêÑ‰∏™feature mapÁöÑÁΩÆ‰ø°Â∫¶ÊçüÂ§±Á≥ªÊï∞

        # loss * bs: Êï¥‰∏™batchÁöÑÊÄªÊçüÂ§±

        # ‰∫éÊçüÂ§±Ëøô‰∏™ÂÄºÊòØ‰∏çÈúÄË¶ÅÊ¢ØÂ∫¶ÂèçÂêë‰º†Êí≠ÁöÑ
