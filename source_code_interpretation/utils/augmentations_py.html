        # Â¶ÇÊûú‰ΩøÁî®ÁöÑÊòØsegmentsÊ†áÁ≠æ(Ê†áÁ≠æ‰∏≠Âê´ÊúâÂ§öËæπÂΩ¢gt)

            # ÊØîÂ¶ÇËØ¥segmentÂùêÊ†áÂè™Êúâ100‰∏™ÔºåÈÄöËøáinterpÂáΩÊï∞Â∞ÜÂÖ∂ÈááÊ†∑‰∏∫n‰∏™(ÈªòËÆ§1000)

        targets = targets[i]

    def bbox_ioa(box1, box2):

        ËÆ°ÁÆóbox1Âíåbox2Áõ∏‰∫§Èù¢ÁßØ‰∏ébox2Èù¢ÁßØÁöÑÊØî‰æã
        Returns the intersection over box2 area given box1, box2. box1 is 4, box2 is nx4. boxes are x1y1x2y2
        :params box1: ‰º†ÂÖ•ÈöèÊú∫ÁîüÊàêÂô™Â£∞ box  [4] = [x1y1x2y2]

        b2_x1, b2_y1, b2_x2, b2_y2 = box2[0], box2[1], box2[2], box2[3]

            ioa = bbox_ioa(box, labels[:, 1:5])

            labels = labels[ioa < 0.60]

    mixupÊï∞ÊçÆÂ¢ûÂº∫, ÊåâÊØî‰æãËûçÂêà‰∏§Âº†ÂõæÁâá  Applies MixUp augmentation
