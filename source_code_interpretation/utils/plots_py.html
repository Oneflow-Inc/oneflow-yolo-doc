    def save_one_box(
    xyxy, im, file=Path("im.jpg"), gain=1.02, pad=10, square=False, BGR=False, save=True
):
    """Áî®Âú®detect.pyÊñá‰ª∂‰∏≠  Áî±optÁöÑsave-cropÂèÇÊï∞ÊéßÂà∂ÊòØÂê¶ÊâßË°å
    Â∞ÜÈ¢ÑÊµãÂà∞ÁöÑÁõÆÊ†á‰ªéÂéüÂõæ‰∏≠Êâ£Âá∫Êù• Ââ™ÂàáÂ•Ω Âπ∂‰øùÂ≠ò ‰ºöÂú® runs/detect/expn ‰∏ãÁîüÊàêcropsÊñá‰ª∂, Â∞ÜÂâ™ÂàáÁöÑÂõæÁâá‰øùÂ≠òÂú®ÈáåÈù¢
    Save image crop as {file} with crop size multiple {gain} and {pad} pixels. Save and/or return crop
    :params xyxy: È¢ÑÊµãÂà∞ÁöÑÁõÆÊ†áÊ°Ü‰ø°ÊÅØ list 4‰∏™tensor x1 y1 x2 y2 Â∑¶‰∏äËßí + Âè≥‰∏ãËßí
    :params im: ÂéüÂõæÁâá ÈúÄË¶ÅË£ÅÂâ™ÁöÑÊ°Ü‰ªéËøô‰∏™ÂéüÂõæ‰∏äË£ÅÂâ™  nparray  (1080, 810, 3)
    :params file: runs\detect\exp\crops\dog\bus.jpg
    :params gain: 1.02 xyxyÁº©ÊîæÂõ†Â≠ê
    :params pad: xyxy pad‰∏ÄÁÇπÁÇπËæπÁïåÊ°Ü Ë£ÅÂâ™Âá∫Êù•‰ºöÊõ¥Â•ΩÁúã
    :params square: ÊòØÂê¶ÈúÄË¶ÅÂ∞ÜxyxyÊîæÁº©ÊàêÊ≠£ÊñπÂΩ¢
    :params BGR: ‰øùÂ≠òÁöÑÂõæÁâáÊòØBGRËøòÊòØRGB „ÄÇ(‰ªÄ‰πàÊòØRGBÊ®°Âºè‰∏éBGRÊ®°ÂºèËØ∑ÂèÇÈòÖËØª https://blog.csdn.net/SGchi/article/details/104474976)
    :params save: ÊòØÂê¶Ë¶Å‰øùÂ≠òÂâ™ÂàáÁöÑÁõÆÊ†áÊ°Ü
    """
    # Save image crop as {file} with crop size multiple {gain} and {pad} pixels. Save and/or return crop
    # viewÂáΩÊï∞: Returns a new tensor with the same data as the self tensor but of a different shape.
    # https://pytorch.org/docs/stable/generated/torch.Tensor.view.html?highlight=torch+view#torch.Tensor.view
    # ÂÖ∂ÂÆûÊåâÁÖßÊàë‰ª¨ÈÄö‰øóÁöÑÁêÜËß£Â∞±ÊòØreshapeÔºå
    # Âè™‰∏çËøáËøôÈáåÊòØreshapeÁöÑÊòØÂº†ÈáèÔºå‰πüÂ∞±ÊòØÂ∞ÜÂº†ÈáèÈáçÊñ∞Ë∞ÉÊï¥‰∏∫Ëá™Â∑±ÊÉ≥Ë¶ÅÁöÑÁª¥Â∫¶ÔºàÂΩ¢Áä∂Â§ßÂ∞èÔºâ
    xyxy = flow.tensor(xyxy).view(-1, 4)  # list -> Tensor [1, 4] = [x1 y1 x2 y2]
    #   OpenCV‰∏≠ÁöÑÂùêÊ†áÁ≥ªÂÆö‰πâÔºåÂ¶Ç‰∏ãÂõæÊâÄÁ§∫:
    #   (0,0)o_________width______________x
    #        |                            |
    #        height                       |
    #        |                            |
    #        |                            | 
    #        |                            |
    #        y____________________________o(w,h)
    # xyxy to xywh [1, 4] = [x y w h]
    b = xyxy2xywh(xyxy)  # boxes
    if square:
        b[:, 2:] = b[:, 2:].max(1)[0].unsqueeze(1)  # attempt rectangle to square
    # box wh * gain + pad  box*gainÂÜçÂä†ÁÇπpad Ë£ÅÂâ™Âá∫Êù•Ê°ÜÊõ¥Â•ΩÁúã
    b[:, 2:] = b[:, 2:] * gain + pad  # box wh * gain + pad
    xyxy = xywh2xyxy(b).long()
    # Clip bounding xyxy bounding boxes to image shape (height, width)
    # Â∞ÜboxesÁöÑÂùêÊ†á(x1y1x2y2 Â∑¶‰∏äËßíÂè≥‰∏ãËßí)ÈôêÂÆöÂú®ÂõæÂÉèÁöÑÂ∞∫ÂØ∏(img_shape hw)ÂÜÖ
    clip_coords(xyxy, im.shape)
    # crop: Ââ™ÂàáÁöÑÁõÆÊ†áÊ°Ühw
    crop = im[
        int(xyxy[0, 1]) : int(xyxy[0, 3]),
        int(xyxy[0, 0]) : int(xyxy[0, 2]),
        :: (1 if BGR else -1),
    ]
    if save:
        file.parent.mkdir(parents=True, exist_ok=True)  # make directory
        f = str(increment_path(file).with_suffix(".jpg"))
        # cv2.imwrite(f, crop)  # save BGR, https://github.com/ultralytics/yolov5/issues/7007 chroma subsampling issue
        # ‰øùÂ≠òÂâ™ÂàáÁöÑÁõÆÊ†áÊ°Ü
        Image.fromarray(crop[..., ::-1],mode="RGB").save(f, quality=95, subsampling=0)  # save RGB
    return crop
