    def resample_segments(segments, n=1000):
    """Áî®Âú®augmentations.pyÊñá‰ª∂‰∏≠ÁöÑrandom_perspectiveÂáΩÊï∞‰∏≠

    :params n: ÈááÊ†∑‰∏™Êï∞
    :return segments: [N, n/2, 2]
    """
    # Up-sample an (n,2) segment
    for i, s in enumerate(segments):
        s = np.concatenate((s, s[0:1, :]), axis=0)
        x = np.linspace(0, len(s) - 1, n)
        xp = np.arange(len(s))
        # ÂØπÊâÄÊúâÁöÑsegmentsÈÉΩËøõË°åÈáçÊñ∞ÈááÊ†∑ ÊØîÂ¶ÇËØ¥segmentÂùêÊ†áÂè™Êúâ100‰∏™ÔºåÈÄöËøáinterpÂáΩÊï∞Â∞ÜÂÖ∂ÈááÊ†∑‰∏∫n‰∏™(ÈªòËÆ§1000)

    return segments
