    def check_anchor_order(m):
    """Áî®Âú®check_anchorsÂáΩÊï∞ÁöÑÊúÄÂêé Á°ÆÂÆö anchors Âíå stride ÁöÑÈ°∫Â∫èÊòØ‰∏ÄËá¥ÁöÑ
    Check anchor order against stride order for YOLOv5 Detect() module m, and correct if necessary
    :params m: model‰∏≠ÁöÑÊúÄÂêé‰∏ÄÂ±Ç DetectÂ±Ç
    """
    # Check anchor order against stride order for YOLOv5 Detect() module m, and correct if necessary
    # ËÆ°ÁÆóanchorÁöÑÈù¢ÁßØ anchor area [9]
    a = m.anchors.prod(-1).mean(-1).view(-1)  # mean anchor area per output layer
    # ËÆ°ÁÆóÊúÄÂ§ßanchor‰∏éÊúÄÂ∞èanchorÈù¢ÁßØÂ∑Æ
    da = a[-1] - a[0]  # delta a
    # ËÆ°ÁÆóÊúÄÂ§ßstride‰∏éÊúÄÂ∞èstrideÂ∑Æ
    # m.stride: model strides 
    # https://github.com/Oneflow-Inc/one-yolov5/blob/bf8c66e011fcf5b8885068074ffc6b56c113a20c/models/yolo.py#L144-L152
    ds = m.stride[-1] - m.stride[0]  # delta s
    # flow.sign(x):ÂΩìxÂ§ß‰∫é/Â∞è‰∫é0Êó∂ÔºåËøîÂõû1/-1
    # Â¶ÇÊûúËøôÈáåanchor‰∏éstrideÈ°∫Â∫è‰∏ç‰∏ÄËá¥ÔºåÂàôÈáçÊñ∞Ë∞ÉÊï¥È°∫Â∫èÔºå‰ΩÜÊ≥®ÊÑèËøôÈáåË¶ÅÊäõÂá∫warning
    if da and (da.sign() != ds.sign()):  # same order
        LOGGER.info(f"{PREFIX}Reversing anchor order")
        m.anchors[:] = m.anchors.flip(0)

    def kmean_anchors(path='./data/coco128.yaml', n=9, img_size=640, thr=4.0, gen=1000, verbose=True):
    """Âú®check_anchors‰∏≠Ë∞ÉÁî®
    ‰ΩøÁî®K-means + ÈÅó‰º†ÁÆóÊ≥ï ÁÆóÂá∫Êõ¥Á¨¶ÂêàÂΩìÂâçÊï∞ÊçÆÈõÜÁöÑanchors
    Creates kmeans-evolved anchors from training dataset
    :params path: Êï∞ÊçÆÈõÜÁöÑË∑ØÂæÑ/Êï∞ÊçÆÈõÜÊú¨Ë∫´
    :params n: anchors ÁöÑ‰∏™Êï∞
    :params img_size: Êï∞ÊçÆÈõÜÂõæÁâáÁ∫¶ÂÆöÁöÑÂ§ßÂ∞è
    :params thr: ÈòàÂÄº Áî± hyp['anchor_t'] ÂèÇÊï∞ÊéßÂà∂
    :params gen: ÈÅó‰º†ÁÆóÊ≥ïËøõÂåñËø≠‰ª£ÁöÑÊ¨°Êï∞(Á™ÅÂèò + ÈÄâÊã©)
    :params verbose: ÊòØÂê¶ÊâìÂç∞ÊâÄÊúâÁöÑËøõÂåñ(ÊàêÂäüÁöÑ)ÁªìÊûú ÈªòËÆ§‰º†ÂÖ•ÊòØFalse, Âè™ÊâìÂç∞ÊúÄ‰Ω≥ÁöÑËøõÂåñÁªìÊûú
    :return k: K-means + ÈÅó‰º†ÁÆóÊ≥ïËøõÂåñÂêéÁöÑanchors
    """
    from scipy.cluster.vq import kmeans


    # Ê≥®ÊÑè‰∏Ä‰∏ã‰∏ãÈù¢ÁöÑthr‰∏çÊòØ‰º†ÂÖ•ÁöÑthrÔºåËÄåÊòØ1/thr, ÊâÄ‰ª•Âú®ËÆ°ÁÆóÊåáÊ†áËøôÊñπÈù¢ËøòÊòØÂíåcheck_anchor‰∏ÄÊ†∑
    thr = 1. / thr  # 0.25
    prefix = colorstr('autoanchor: ')

    def metric(k, wh):  # compute metrics
        """Áî®‰∫é print_results ÂáΩÊï∞Âíå anchor_fitness ÂáΩÊï∞
        ËÆ°ÁÆóratio metric: Êï¥‰∏™Êï∞ÊçÆÈõÜÁöÑ  ground truth Ê°Ü‰∏é anchor ÂØπÂ∫îÂÆΩÊØîÂíåÈ´òÊØîÂç≥:gt_w/k_w,gt_h/k_h + x + best_x  Áî®‰∫éÂêéÁª≠ËÆ°ÁÆóBPR+aat
        Ê≥®ÊÑèÊàë‰ª¨ËøôÈáåÈÄâÊã©ÁöÑmetricÊòØ ground truth Ê°Ü‰∏éanchorÂØπÂ∫îÂÆΩÊØîÂíåÈ´òÊØî ËÄå‰∏çÊòØÂ∏∏Áî®ÁöÑiou ËøôÁÇπ‰πü‰∏énmsÁöÑÁ≠õÈÄâÊù°‰ª∂ÂØπÂ∫î ÊòØyolov5‰∏≠‰ΩøÁî®ÁöÑÊñ∞ÊñπÊ≥ï
        :params k: anchorÊ°Ü
        :params wh: Êï¥‰∏™Êï∞ÊçÆÈõÜÁöÑ wh [N, 2]
        :return x: [N, 9] N ‰∏™ ground truth Ê°Ü‰∏éÊâÄÊúâ anchor Ê°ÜÁöÑÂÆΩÊØîÊàñÈ´òÊØî(‰∏§ËÄÖ‰πã‰∏≠ËæÉÂ∞èËÄÖ)
        :return x.max(1)[0]: [N] N‰∏™ ground truth Ê°Ü‰∏éÊâÄÊúâ anchor Ê°Ü‰∏≠ÁöÑÊúÄÂ§ßÂÆΩÊØîÊàñÈ´òÊØî(‰∏§ËÄÖ‰πã‰∏≠ËæÉÂ∞èËÄÖ)
        """
        # [N, 1, 2] / [1, 9, 2] = [N, 9, 2]  N‰∏™gt_whÂíå9‰∏™anchorÁöÑk_whÂÆΩÊØîÂíåÈ´òÊØî
        # ‰∏§ËÄÖÁöÑÈáçÂêàÁ®ãÂ∫¶Ë∂äÈ´ò Â∞±Ë∂äË∂ãËøë‰∫é1 ËøúÁ¶ª1(<1 Êàñ >1)ÈáçÂêàÁ®ãÂ∫¶ÈÉΩË∂ä‰Ωé
        r = wh[:, None] / k[None]
        # r=gt_height/anchor_height  gt_width / anchor_width  ÊúâÂèØËÉΩÂ§ß‰∫é1Ôºå‰πüÂèØËÉΩÂ∞è‰∫éÁ≠â‰∫é1
        # flow.min(r, 1. / r): [N, 9, 2] Â∞ÜÊâÄÊúâÁöÑÂÆΩÊØîÂíåÈ´òÊØîÁªü‰∏ÄÂà∞ <=1
        # .min(2): value=[N, 9] ÈÄâÂá∫ÊØè‰∏™ ground truth ‰∏™Âíå anchor ÁöÑÂÆΩÊØîÂíåÈ´òÊØîÊúÄÂ∞èÁöÑÂÄº   index: [N, 9] Ëøô‰∏™ÊúÄÂ∞èÂÄºÊòØÂÆΩÊØî(0)ËøòÊòØÈ´òÊØî(1)
        # [0] ËøîÂõû value [N, 9]  ÊØè‰∏™ ground truth ‰∏™Âíå anchor ÁöÑÂÆΩÊØîÂíåÈ´òÊØîÊúÄÂ∞èÁöÑÂÄº Â∞±ÊòØÊâÄÊúâ ground truth ‰∏é anchor ÈáçÂêàÁ®ãÂ∫¶ÊúÄ‰ΩéÁöÑ
        x = flow.min(r, 1. / r).min(2)[0]  # ratio metric
        # x = wh_iou(wh, flow.tensor(k))  # IoU metric
        # x.max(1)[0]: [N] ËøîÂõûÊØè‰∏™ ground truth ÂíåÊâÄÊúâ anchor(9‰∏™) ‰∏≠ÂÆΩÊØî/È´òÊØîÊúÄÂ§ßÁöÑÂÄº
        return x, x.max(1)[0]  # x, best_x

    def anchor_fitness(k):   # mutation fitness
        """Áî®‰∫é kmean_anchors ÂáΩÊï∞
        ÈÄÇÂ∫îÂ∫¶ËÆ°ÁÆó ‰ºòËÉúÂä£Ê±∞ Áî®‰∫éÈÅó‰º†ÁÆóÊ≥ï‰∏≠Ë°°ÈáèÁ™ÅÂèòÊòØÂê¶ÊúâÊïàÁöÑÊ†áÊ≥® Â¶ÇÊûúÊúâÊïàÂ∞±ËøõË°åÈÄâÊã©Êìç‰ΩúÔºåÊó†ÊïàÂ∞±ÁªßÁª≠‰∏ã‰∏ÄËΩÆÁöÑÁ™ÅÂèò
        :params k: [9, 2] K-meansÁîüÊàêÁöÑ 9 ‰∏™anchors     wh: [N, 2]: Êï∞ÊçÆÈõÜÁöÑÊâÄÊúâ ground truth Ê°ÜÁöÑÂÆΩÈ´ò
        :return (best * (best > thr).float()).mean()=ÈÄÇÂ∫îÂ∫¶ËÆ°ÁÆóÂÖ¨Âºè [1] Ê≥®ÊÑèÂíåBPRÊúâÂå∫Âà´ ËøôÈáåÊòØËá™ÂÆö‰πâÁöÑ‰∏ÄÁßçÈÄÇÂ∫îÂ∫¶ÂÖ¨Âºè
                ËøîÂõûÁöÑÊòØËæìÂÖ•Ê≠§Êó∂anchor k ÂØπÂ∫îÁöÑÈÄÇÂ∫îÂ∫¶
        """
        _, best = metric(flow.tensor(k, dtype=flow.float32), wh)
        return (best * (best > thr).float()).mean()  # fitness

    def print_results(k):
        """Áî®‰∫é kmean_anchors ÂáΩÊï∞‰∏≠ÊâìÂç∞K-meansËÆ°ÁÆóÁõ∏ÂÖ≥‰ø°ÊÅØ
        ËÆ°ÁÆóBPR„ÄÅaat=>ÊâìÂç∞‰ø°ÊÅØ: ÈòàÂÄº+BPR+aat  anchor‰∏™Êï∞+ÂõæÁâáÂ§ßÂ∞è+metric_all+best_mean+past_mean+KmeansËÅöÁ±ªÂá∫Êù•ÁöÑanchorÊ°Ü(ÂõõËàç‰∫îÂÖ•)
        :params k: K-meansÂæóÂà∞ÁöÑanchor k
        :return k: input
        """
        # Â∞ÜK-meansÂæóÂà∞ÁöÑanchor kÊåâÈù¢ÁßØ‰ªéÂ∞èÂà∞Â§ßÊéíÂ∫è
        k = k[np.argsort(k.prod(1))]
        # x: [N, 9] N‰∏™ ground truth Ê°Ü‰∏éÊâÄÊúâanchorÊ°ÜÁöÑÂÆΩÊØîÊàñÈ´òÊØî(‰∏§ËÄÖ‰πã‰∏≠ËæÉÂ∞èËÄÖ)
        # best: [N] N‰∏™ ground truth Ê°Ü‰∏éÊâÄÊúâanchorÊ°Ü‰∏≠ÁöÑÊúÄÂ§ßÂÆΩÊØîÊàñÈ´òÊØî(‰∏§ËÄÖ‰πã‰∏≠ËæÉÂ∞èËÄÖ)
        x, best = metric(k, wh0)
        # (best > thr).float(): True=>1.  False->0.  .mean(): Ê±ÇÂùáÂÄº
        # BPR(best possible recall): ÊúÄÂ§öËÉΩË¢´Âè¨Âõû(ÈÄöËøáthr)ÁöÑ ground truth Ê°ÜÊï∞Èáè / ÊâÄÊúâ ground truth Ê°ÜÊï∞Èáè  [1] 0.96223  Â∞è‰∫é0.98 Êâç‰ºöÁî®K-meansËÆ°ÁÆóanchor
        # aat(anchors above threshold): [1] 3.54360 ÊØè‰∏™targetÂπ≥ÂùáÊúâÂ§öÂ∞ë‰∏™anchors
        BPR, aat = (best > thr).float().mean(), (x > thr).float().mean() * n  # best possible recall, anch > thr
        f = anchor_fitness(k)
        # print(f'{prefix}thr={thr:.2f}: {BPR:.4f} best possible recall, {aat:.2f} anchors past thr')
        # print(f'{prefix}n={n}, img_size={img_size}, metric_all={x.mean():.3f}/{best.mean():.3f}-mean/best, '
        #       f'past_thr={x[x > thr].mean():.3f}-mean: ', end='')
        print(f"aat: {aat:.5f}, fitness: {f:.5f}, best possible recall: {BPR:.5f}")
        for i, x in enumerate(k):
            print('%i,%i' % (round(x[0]), round(x[1])), end=',  ' if i < len(k) - 1 else '\n')  # use in *.cfg

        return k


    # ËΩΩÂÖ•Êï∞ÊçÆÈõÜ
    if isinstance(path, str):  # *.yaml file
        with open(path) as f:
            data_dict = yaml.safe_load(f)  # model dict
        from utils.datasets import LoadImagesAndLabels
        dataset = LoadImagesAndLabels(data_dict['train'], augment=True, rect=True)
    else:
        dataset = path  # dataset

    # ÂæóÂà∞Êï∞ÊçÆÈõÜ‰∏≠ÊâÄÊúâÊï∞ÊçÆÁöÑ wh
    # Â∞ÜÊï∞ÊçÆÈõÜÂõæÁâáÁöÑÊúÄÈïøËæπÁº©ÊîæÂà∞ img_size, ËæÉÂ∞èËæπÁõ∏Â∫îÁº©Êîæ
    shapes = img_size * dataset.shapes / dataset.shapes.max(1, keepdims=True)
    # Â∞ÜÂéüÊú¨Êï∞ÊçÆÈõÜ‰∏≠gt boxesÂΩí‰∏ÄÂåñÁöÑwhÁº©ÊîæÂà∞shapesÂ∞∫Â∫¶
    wh0 = np.concatenate([l[:, 3:5] * s for s, l in zip(shapes, dataset.labels)])

    # ÁªüËÆ°gt boxes‰∏≠ÂÆΩÊàñËÄÖÈ´òÂ∞è‰∫é 3 ‰∏™ÂÉèÁ¥†ÁöÑ‰∏™Êï∞, ÁõÆÊ†áÂ§™Â∞è ÂèëÂá∫Ë≠¶Âëä
    i = (wh0 < 3.0).any(1).sum()
    if i:
        print(f'{prefix}WARNING: Extremely small objects found. {i} of {len(wh0)} labels are < 3 pixels in size.')

    # Á≠õÈÄâÂá∫ label Â§ß‰∫é 2 ‰∏™ÂÉèÁ¥†ÁöÑÊ°ÜÊãøÊù•ËÅöÁ±ª, [...]ÂÜÖÁöÑÁõ∏ÂΩì‰∫é‰∏Ä‰∏™Á≠õÈÄâÂô®, ‰∏∫TrueÁöÑÁïô‰∏ã
    wh = wh0[(wh0 >= 2.0).any(1)]  # filter > 2 pixels
    # wh = wh * (np.random.rand(wh.shape[0], 1) * 0.9 + 0.1)  # multiply by random scale 0-1

    # KmeansËÅöÁ±ªÊñπÊ≥ï: ‰ΩøÁî®Ê¨ßÂºèË∑ùÁ¶ªÊù•ËøõË°åËÅöÁ±ª
    print(f'{prefix}Running kmeans for {n} anchors on {len(wh)} gt boxes...')
    # ËÆ°ÁÆóÂÆΩÂíåÈ´òÁöÑÊ†áÂáÜÂ∑Æ->[w_std,h_std]
    s = wh.std(0)  # sigmas for whitening
    # ÂºÄÂßãËÅöÁ±ª,‰ªçÁÑ∂ÊòØËÅöÊàê n Á±ª,ËøîÂõûËÅöÁ±ªÂêéÁöÑanchors k(Ëøô‰∏™anchors kÊòØÁôΩÂåñÂêéÊï∞ÊçÆÁöÑanchorÊ°Üs)
    # Âè¶Â§ñËøòË¶ÅÊ≥®ÊÑèÁöÑÊòØËøôÈáåÁöÑkmeans‰ΩøÁî®Ê¨ßÂºèË∑ùÁ¶ªÊù•ËÆ°ÁÆóÁöÑ
    # ËøêË°åK-meansÁöÑÊ¨°Êï∞‰∏∫30Ê¨°  obs: ‰º†ÂÖ•ÁöÑÊï∞ÊçÆÂøÖÈ°ªÂÖàÁôΩÂåñÂ§ÑÁêÜ 'whiten operation'
    # ÁôΩÂåñÂ§ÑÁêÜ: Êñ∞Êï∞ÊçÆÁöÑÊ†áÂáÜÂ∑Æ=1 Èôç‰ΩéÊï∞ÊçÆ‰πãÈó¥ÁöÑÁõ∏ÂÖ≥Â∫¶Ôºå‰∏çÂêåÊï∞ÊçÆÊâÄËï¥Âê´ÁöÑ‰ø°ÊÅØ‰πãÈó¥ÁöÑÈáçÂ§çÊÄßÂ∞±‰ºöÈôç‰ΩéÔºåÁΩëÁªúÁöÑËÆ≠ÁªÉÊïàÁéáÂ∞±‰ºöÊèêÈ´ò
    # ÁôΩÂåñÊìç‰ΩúÂèÇËÄÉÂçöÂÆ¢: https://blog.csdn.net/weixin_37872766/article/details/102957235
    k, dist = kmeans(wh / s, n, iter=30)  # points, mean distance
    assert len(k) == n, print(f'{prefix}ERROR: scipy.cluster.vq.kmeans requested {n} points but returned only {len(k)}')
    k *= s  # k*s ÂæóÂà∞ÂéüÊù•Êï∞ÊçÆ(ÁôΩÂåñÂâç)ÁöÑ anchor Ê°Ü

    wh = flow.tensor(wh, dtype=flow.float32)  # filtered wh
    wh0 = flow.tensor(wh0, dtype=flow.float32)  # unfiltered wh0

    # ËæìÂá∫Êñ∞ÁÆóÁöÑanchors k Áõ∏ÂÖ≥ÁöÑ‰ø°ÊÅØ
    k = print_results(k)

    # Plot wh
    # k, d = [None] * 20, [None] * 20
    # for i in tqdm(range(1, 21)):
    #     k[i-1], d[i-1] = kmeans(wh / s, i)  # points, mean distance
    # fig, ax = plt.subplots(1, 2, figsize=(14, 7), tight_layout=True)
    # ax = ax.ravel()
    # ax[0].plot(np.arange(1, 21), np.array(d) ** 2, marker='.')
    # fig, ax = plt.subplots(1, 2, figsize=(14, 7))  # plot wh
    # ax[0].hist(wh[wh[:, 0]<100, 0], 400)
    # ax[1].hist(wh[wh[:, 1]<100, 1], 400)
    # fig.savefig('wh.png', dpi=200)

    # Evolve Á±ª‰ººÈÅó‰º†/ËøõÂåñÁÆóÊ≥ï  ÂèòÂºÇÊìç‰Ωú
    npr = np.random   # ÈöèÊú∫Â∑•ÂÖ∑
    # f: fitness 0.62690
    # sh: (9,2)
    # mp: Á™ÅÂèòÊØî‰æãmutation prob=0.9   s: sigma=0.1
    f, sh, mp, s = anchor_fitness(k), k.shape, 0.9, 0.1  # fitness, generations, mutation prob, sigma
    pbar = tqdm(range(gen), desc=f'{prefix}Evolving anchors with Genetic Algorithm:')  # progress bar
    # Ê†πÊçÆËÅöÁ±ªÂá∫Êù•ÁöÑn‰∏™ÁÇπÈááÁî®ÈÅó‰º†ÁÆóÊ≥ïÁîüÊàêÊñ∞ÁöÑanchor
    for _ in pbar:
        # ÈáçÂ§ç1000Ê¨°Á™ÅÂèò+ÈÄâÊã© ÈÄâÊã©Âá∫1000Ê¨°Á™ÅÂèòÈáåÁöÑÊúÄ‰Ω≥anchor kÂíåÊúÄ‰Ω≥ÈÄÇÂ∫îÂ∫¶f
        v = np.ones(sh)  # v [9, 2] ÂÖ®ÊòØ1
        while (v == 1).all():
            # ‰∫ßÁîüÂèòÂºÇËßÑÂàô mutate until a change occurs (prevent duplicates)
            # npr.random(sh) < mp: ËÆ©v‰ª•90%ÁöÑÊØî‰æãËøõË°åÂèòÂºÇ  ÈÄâÂà∞ÂèòÂºÇÁöÑÂ∞±‰∏∫1  Ê≤°ÊúâÈÄâÂà∞ÂèòÂºÇÁöÑÂ∞±‰∏∫0
            v = ((npr.random(sh) < mp) * npr.random() * npr.randn(*sh) * s + 1).clip(0.3, 3.0)
        # ÂèòÂºÇ(ÊîπÂèòËøô‰∏ÄÊó∂Âàª‰πãÂâçÁöÑÊúÄ‰Ω≥ÈÄÇÂ∫îÂ∫¶ÂØπÂ∫îÁöÑanchor k)
        kg = (k.copy() * v).clip(min=2.0)
        # ËÆ°ÁÆóÂèòÂºÇÂêéÁöÑanchor kgÁöÑÈÄÇÂ∫îÂ∫¶
        fg = anchor_fitness(kg)
        # Â¶ÇÊûúÂèòÂºÇÂêéÁöÑanchor kgÁöÑÈÄÇÂ∫îÂ∫¶>ÊúÄ‰Ω≥ÈÄÇÂ∫îÂ∫¶k Â∞±ËøõË°åÈÄâÊã©Êìç‰Ωú
        if fg > f:
            # ÈÄâÊã©ÂèòÂºÇÂêéÁöÑanchor kg‰∏∫ÊúÄ‰Ω≥ÁöÑanchor k ÂèòÂºÇÂêéÁöÑÈÄÇÂ∫îÂ∫¶fg‰∏∫ÊúÄ‰Ω≥ÈÄÇÂ∫îÂ∫¶f
            f, k = fg, kg.copy()

            # ÊâìÂç∞‰ø°ÊÅØ
            pbar.desc = f'{prefix}Evolving anchors with Genetic Algorithm: fitness = {f:.4f}'
            if verbose:
                print_results(k)
    return print_results(k)

    # Check anchor fit to data, recompute if necessary  
    """Áî®‰∫étrain.py‰∏≠
    ÈÄöËøáBPRÁ°ÆÂÆöÊòØÂê¶ÈúÄË¶ÅÊîπÂèòanchors ÈúÄË¶ÅÂ∞±Ë∞ÉÁî®K-meansÈáçÊñ∞ËÆ°ÁÆóanchors
    Check anchor fit to data, recompute if necessary
    :params dataset: Ëá™ÂÆö‰πâÊï∞ÊçÆÈõÜLoadImagesAndLabelsËøîÂõûÁöÑÊï∞ÊçÆÈõÜ
    :params model: ÂàùÂßãÂåñÁöÑÊ®°Âûã
    :params thr: Ë∂ÖÂèÇ‰∏≠ÂæóÂà∞  ÁïåÂÆöanchor‰∏élabelÂåπÈÖçÁ®ãÂ∫¶ÁöÑÈòàÂÄº
    :params imgsz: ÂõæÁâáÂ∞∫ÂØ∏ ÈªòËÆ§640
    """
    # ‰ªémodel‰∏≠ÂèñÂá∫ÊúÄÂêé‰∏ÄÂ±Ç(Detect)
    m = model.module.model[-1] if hasattr(model, "module") else model.model[-1]  # Detect()
    # dataset.shapes.max(1, keepdims=True) = ÊØèÂº†ÂõæÁâáÁöÑËæÉÈïøËæπ
    # shapes: Â∞ÜÊï∞ÊçÆÈõÜÂõæÁâáÁöÑÊúÄÈïøËæπÁº©ÊîæÂà∞img_size, ËæÉÂ∞èËæπÁõ∏Â∫îÁº©Êîæ ÂæóÂà∞Êñ∞ÁöÑÊâÄÊúâÊï∞ÊçÆÈõÜÂõæÁâáÁöÑÂÆΩÈ´ò [N, 2]
    shapes = imgsz * dataset.shapes / dataset.shapes.max(1, keepdims=True)
    # ‰∫ßÁîüÈöèÊú∫Êï∞scale [img_size, 1]
    scale = np.random.uniform(0.9, 1.1, size=(shapes.shape[0], 1))  # augment scale
    # [6301, 2]  ÊâÄÊúâtarget(6301‰∏™)ÁöÑwh   Âü∫‰∫éÂéüÂõæÂ§ßÂ∞è    shapes * scale: ÈöèÊú∫ÂåñÂ∞∫Â∫¶ÂèòÂåñ
    wh = flow.tensor(np.concatenate([l[:, 3:5] * s for s, l in zip(shapes * scale, dataset.labels)])).float()  # wh

    def metric(k):  # compute metric
        """Áî®Âú®check_anchorsÂáΩÊï∞‰∏≠  compute metric
        Ê†πÊçÆÊï∞ÊçÆÈõÜÁöÑÊâÄÊúâÂõæÁâáÁöÑwhÂíåÂΩìÂâçÊâÄÊúâanchors kËÆ°ÁÆó BPR(best possible recall) Âíå aat(anchors above threshold)
        :params k: anchors [9, 2]  wh: [N, 2]
        :return BPR: best possible recall ÊúÄÂ§öËÉΩË¢´Âè¨Âõû(ÈÄöËøáthr)ÁöÑ ground truth Ê°ÜÊï∞Èáè / ÊâÄÊúâ ground truth Ê°ÜÊï∞ÈáèÂ∞è‰∫é0.98 Êâç‰ºöÁî®K-meansËÆ°ÁÆóanchor
        :return aat: anchors above threshold ÊØè‰∏™targetÂπ≥ÂùáÊúâÂ§öÂ∞ë‰∏™anchors
        """
        # NoneÊ∑ªÂä†Áª¥Â∫¶  ÊâÄÊúâtarget(gt)ÁöÑwh wh[:, None] [6301, 2]->[6301, 1, 2]
        #             ÊâÄÊúâanchorÁöÑwh k[None] [9, 2]->[1, 9, 2]
        # r: targetÁöÑÈ´òhÂÆΩw‰∏éanchorÁöÑÈ´òh_aÂÆΩw_aÁöÑÊØîÂÄºÔºåÂç≥h/h_a, w/w_a  [6301, 9, 2]  ÊúâÂèØËÉΩÂ§ß‰∫é1Ôºå‰πüÂèØËÉΩÂ∞è‰∫éÁ≠â‰∫é1
        r = wh[:, None] / k[None]
        # x È´òÂÆΩÊØîÂíåÂÆΩÈ´òÊØîÁöÑÊúÄÂ∞èÂÄº Êó†ËÆ∫rÂ§ß‰∫é1ÔºåËøòÊòØÂ∞è‰∫éÁ≠â‰∫é1ÊúÄÂêéÁªü‰∏ÄÁªìÊûúÈÉΩË¶ÅÂ∞è‰∫é1   [6301, 9]
        x = flow.min(r, 1 / r).min(2)[0]  # ratio metric
        # best [6301] ‰∏∫ÊØè‰∏™ ground truth Ê°ÜÈÄâÊã©ÂåπÈÖçÊâÄÊúâanchorsÂÆΩÈ´òÊØî‰æãÂÄºÊúÄÂ•ΩÁöÑÈÇ£‰∏Ä‰∏™ÊØîÂÄº
        best = x.max(1)[0]  # best_x
        # aat(anchors above threshold)  ÊØè‰∏™targetÂπ≥ÂùáÊúâÂ§öÂ∞ë‰∏™anchors
        aat = (x > 1 / thr).float().sum(1).mean()  # anchors above threshold
        # BPR(best possible recall) = ÊúÄÂ§öËÉΩË¢´Âè¨Âõû(ÈÄöËøáthr)ÁöÑ ground truth Ê°ÜÊï∞Èáè / ÊâÄÊúâ ground truth Ê°ÜÊï∞Èáè   Â∞è‰∫é0.98 Êâç‰ºöÁî®K-meansËÆ°ÁÆóanchor
        BPR = (best > 1 / thr).float().mean()  # best possible recall
        return BPR, aat

    stride = m.stride.to(m.anchors.device).view(-1, 1, 1)  # model strides
    # anchors: [N,2]  ÊâÄÊúâanchorsÁöÑÂÆΩÈ´ò   Âü∫‰∫éÁº©ÊîæÂêéÁöÑÂõæÁâáÂ§ßÂ∞è(ËæÉÈïøËæπ‰∏∫640 ËæÉÂ∞èËæπÁõ∏Â∫îÁº©Êîæ)
    anchors = m.anchors.clone() * stride  # current anchors
    BPR, aat = metric(anchors.cpu().view(-1, 2))
    s = f"\n{PREFIX}{aat:.2f} anchors/target, {BPR:.3f} Best Possible Recall (BPR). "
    # ËÄÉËôëËøô9Á±ªanchorÁöÑÂÆΩÈ´òÂíå ground truth Ê°ÜÁöÑÂÆΩÈ´ò‰πãÈó¥ÁöÑÂ∑ÆË∑ù, Â¶ÇÊûúBPR<0.98(ËØ¥ÊòéÂΩìÂâçanchor‰∏çËÉΩÂæàÂ•ΩÁöÑÂåπÈÖçÊï∞ÊçÆÈõÜ ground truth Ê°Ü)Â∞±‰ºöÊ†πÊçÆK-meansÁÆóÊ≥ïÈáçÊñ∞ËÅöÁ±ªÊñ∞ÁöÑanchor
    if BPR > 0.98:  # threshold to recompute
        LOGGER.info(f"{s}Current anchors are a good fit to dataset ‚úÖ")
    else:
        LOGGER.info(f"{s}Anchors are a poor fit to dataset ‚ö†Ô∏è, attempting to improve...")
        na = m.anchors.numel() // 2  # number of anchors
        try:
            # Â¶ÇÊûúBPR<0.98(ÊúÄÂ§ß‰∏∫1 Ë∂äÂ§ßË∂äÂ•Ω) ‰ΩøÁî®K-means + ÈÅó‰º†ËøõÂåñÁÆóÊ≥ïÈÄâÊã©Âá∫‰∏éÊï∞ÊçÆÈõÜÊõ¥ÂåπÈÖçÁöÑanchorsÊ°Ü  [9, 2]

        # ËÆ°ÁÆóÊñ∞ÁöÑanchorsÁöÑnew_BPR
        new_BPR = metric(anchors)[0]
        # ÊØîËæÉ K-means + ÈÅó‰º†ËøõÂåñÁÆóÊ≥ïËøõÂåñÂêéÁöÑanchorsÁöÑnew_BPRÂíåÂéüÂßãanchorsÁöÑBPR
        # Ê≥®ÊÑè: ËøôÈáåÂπ∂‰∏ç‰∏ÄÂÆöËøõÂåñÂêéÁöÑBPRÂøÖÂ§ß‰∫éÂéüÂßãanchorsÁöÑBPR, Âõ†‰∏∫‰∏§ËÄÖÁöÑË°°ÈáèÊ†áÊ≥®ÊòØ‰∏ç‰∏ÄÊ†∑ÁöÑ  ËøõÂåñÁÆóÊ≥ïÁöÑË°°ÈáèÊ†áÂáÜÊòØÈÄÇÂ∫îÂ∫¶ ËÄåËøôÈáåÊØîÁöÑÊòØBPR
        if new_BPR > BPR:  # replace anchors
            anchors = flow.tensor(anchors, device=m.anchors.device).type_as(m.anchors)
            # ÊõøÊç¢mÁöÑanchor_grid                      [9, 2] -> [3, 1, 3, 1, 1, 2]
