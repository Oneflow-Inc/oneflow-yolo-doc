    def _descale_pred(self, p, flips, scale, img_size):
        # de-scale predictions following augmented inference (inverse operation)
        """Áî®Âú®‰∏äÈù¢ÁöÑ__init__ÂáΩÊï∞‰∏ä
        Â∞ÜÊé®ÁêÜÁªìÊûúÊÅ¢Â§çÂà∞ÂéüÂõæÂõæÁâáÂ∞∫ÂØ∏  Test Time Augmentation(TTA)‰∏≠Áî®Âà∞
         de-scale predictions following augmented inference (inverse operation)
        @params p: Êé®ÁêÜÁªìÊûú
        @params flips:
        @params scale:
        @params img_size:
        """
        if self.inplace:
            p[..., :4] /= scale  # de-scale
            if flips == 2:
                p[..., 1] = img_size[0] - p[..., 1]  # de-flip ud
            elif flips == 3:
                p[..., 0] = img_size[1] - p[..., 0]  # de-flip lr
        else:
            x, y, wh = p[..., 0:1] / scale, p[..., 1:2] / scale, p[..., 2:4] / scale  # de-scale
            if flips == 2:
                y = img_size[0] - y  # de-flip ud
            elif flips == 3:
                x = img_size[1] - x  # de-flip lr
            p = flow.cat((x, y, wh, p[..., 4:]), -1)
        return p

    def _clip_augmented(self, y):
        # Clip YOLOv5 augmented inference tails
        nl = self.model[-1].nl  # number of detection layers (P3-P5)
        g = sum(4 ** x for x in range(nl))  # grid points
        e = 1  # exclude layer count
        i = (y[0].shape[1] // g) * sum(4 ** x for x in range(e))  # indices
        y[0] = y[0][:, :-i]  # large
        i = (y[-1].shape[1] // g) * sum(4 ** (nl - 1 - x) for x in range(e))  # indices
        y[-1] = y[-1][:, i:]  # small
        return y

    class Detect(nn.Module):
    """
    DetectÊ®°ÂùóÊòØÁî®Êù•ÊûÑÂª∫DetectÂ±ÇÁöÑÔºåÂ∞ÜËæìÂÖ•feature map ÈÄöËøá‰∏Ä‰∏™Âç∑ÁßØÊìç‰ΩúÂíåÂÖ¨ÂºèËÆ°ÁÆóÂà∞Êàë‰ª¨ÊÉ≥Ë¶ÅÁöÑshape, ‰∏∫ÂêéÈù¢ÁöÑËÆ°ÁÆóÊçüÂ§±ÊàñËÄÖNMSÂêéÂ§ÑÁêÜ‰ΩúÂáÜÂ§á
    """
    stride = None  # strides computed during build
    onnx_dynamic = False  # ONNX export parameter
    export = False  # export mode

    def __init__(self, nc=80, anchors=(), ch=(), inplace=True):  # detection layer
        super().__init__()
        #  nc:ÂàÜÁ±ªÊï∞Èáè
        self.nc = nc  # number of classes  COCO : 80
        #  no:ÊØè‰∏™anchorÁöÑËæìÂá∫Êï∞ COCO: 80 + 5 = 85 
        self.no = nc + 5  # number of outputs per anchor  DetectÁöÑ‰∏™Êï∞ 3
        # nl:È¢ÑÊµãÂ±ÇÊï∞ÔºåÊ≠§Ê¨°‰∏∫3
        self.nl = len(anchors)  # number of detection layers
        #  na:anchorsÁöÑÊï∞ÈáèÔºåÊ≠§Ê¨°‰∏∫3
        self.na = len(anchors[0]) // 2  # number of anchors
        #  grid:Ê†ºÂ≠êÂùêÊ†áÁ≥ªÔºåÂ∑¶‰∏äËßí‰∏∫(1,1),Âè≥‰∏ãËßí‰∏∫(input.w/stride,input.h/stride)
        self.grid = [flow.zeros(1)] * self.nl  # init grid
        self.anchor_grid = [flow.zeros(1)] * self.nl  # init anchor grid
        # ÂÜôÂÖ•ÁºìÂ≠ò‰∏≠ÔºåÂπ∂ÂëΩÂêç‰∏∫anchors
        # register_buffer
        # Ê®°Âûã‰∏≠ÈúÄË¶Å‰øùÂ≠òÁöÑÂèÇÊï∞‰∏ÄËà¨Êúâ‰∏§ÁßçÔºö‰∏ÄÁßçÊòØÂèçÂêë‰º†Êí≠ÈúÄË¶ÅË¢´optimizerÊõ¥Êñ∞ÁöÑÔºåÁß∞‰∏∫parameter; Âè¶‰∏ÄÁßç‰∏çË¶ÅË¢´Êõ¥Êñ∞Áß∞‰∏∫buffer
        # bufferÁöÑÂèÇÊï∞Êõ¥Êñ∞ÊòØÂú®forward‰∏≠ÔºåËÄåoptim.stepÂè™ËÉΩÊõ¥Êñ∞nn.parameterÁ±ªÂûãÁöÑÂèÇÊï∞
        self.register_buffer('anchors', flow.tensor(anchors).float().view(self.nl, -1, 2))  # shape(nl,na,2)
        # Â∞ÜËæìÂá∫ÈÄöËøáÂç∑ÁßØÂà∞ self.no * self.na ÁöÑÈÄöÈÅìÔºåËææÂà∞ÂÖ®ËøûÊé•ÁöÑ‰ΩúÁî®
        self.m = nn.ModuleList(nn.Conv2d(x, self.no * self.na, 1) for x in ch)  # output conv
        self.inplace = inplace  # use inplace ops (e.g. slice assignment)

    def forward(self, x):
        z = []  # inference output
        for i in range(self.nl):
            x[i] = self.m[i](x[i])  # conv
            bs, _, ny, nx = x[i].shape  # x(bs,255,20,20) to x(bs,3,20,20,85)
            x[i] = x[i].view(bs, self.na, self.no, ny, nx).permute(0, 1, 3, 4, 2).contiguous()

            if not self.training:  # inference
                # ÊûÑÈÄ†ÁΩëÊ†º
                # Âõ†‰∏∫Êé®ÁêÜËøîÂõûÁöÑ‰∏çÊòØÂΩí‰∏ÄÂåñÂêéÁöÑÁΩëÊ†ºÂÅèÁßªÈáè ÈúÄË¶ÅÂÜçÂä†‰∏äÁΩëÊ†ºÁöÑ‰ΩçÁΩÆ ÂæóÂà∞ÊúÄÁªàÁöÑÊé®ÁêÜÂùêÊ†á ÂÜçÈÄÅÂÖ•nms
                # ÊâÄ‰ª•ËøôÈáåÊûÑÂª∫ÁΩëÊ†ºÂ∞±ÊòØ‰∏∫‰∫ÜËÆ∞ÂΩïÊØè‰∏™gridÁöÑÁΩëÊ†ºÂùêÊ†á ÊñπÈù¢ÂêéÈù¢‰ΩøÁî®
                if self.onnx_dynamic or self.grid[i].shape[2:4] != x[i].shape[2:4]:
                    # ÂêëÂâç‰º†Êí≠Êó∂ÈúÄË¶ÅÂ∞ÜÁõ∏ÂØπÂùêÊ†áËΩ¨Êç¢Âà∞gridÁªùÂØπÂùêÊ†áÁ≥ª‰∏≠
                    self.grid[i], self.anchor_grid[i] = self._make_grid(nx, ny, i)
                y = x[i].sigmoid()
                if self.inplace:
                    # ÈªòËÆ§ÊâßË°å ‰∏ç‰ΩøÁî®AWS Inferentia
                    # ËøôÈáåÁöÑÂÖ¨ÂºèÂíåyolov3„ÄÅv4‰∏≠‰ΩøÁî®ÁöÑ‰∏ç‰∏ÄÊ†∑ ÊòØyolov5‰ΩúËÄÖËá™Â∑±Áî®ÁöÑ ÊïàÊûúÊõ¥Â•Ω
                    y[..., 0:2] = (y[..., 0:2] * 2 + self.grid[i]) * self.stride[i]  # xy
                    y[..., 2:4] = (y[..., 2:4] * 2) ** 2 * self.anchor_grid[i]  # wh
                else:  # for YOLOv5 on AWS Inferentia https://github.com/ultralytics/yolov5/pull/2953
                    xy, wh, conf = y.split((2, 2, self.nc + 1), 4)  # y.tensor_split((2, 4, 5), 4)  
                    xy = (xy * 2 + self.grid[i]) * self.stride[i]  # xy
                    wh = (wh * 2) ** 2 * self.anchor_grid[i]  # wh
                    y = flow.cat((xy, wh, conf), 4)

                # z [oneflow.Size([1, 19200, 85])  oneflow.Size([1, 4800, 85]) oneflow.Size([1, 1200, 85])]
                z.append(y.view(bs, -1, self.no))


        return x if self.training else (flow.cat(z, 1),) if self.export else (flow.cat(z, 1), x)

    # Áõ∏ÂØπÂùêÊ†áËΩ¨Êç¢Âà∞gridÁªùÂØπÂùêÊ†áÁ≥ª
    def _make_grid(self, nx=20, ny=20, i=0):
        d = self.anchors[i].device
        t = self.anchors[i].dtype
        shape = 1, self.na, ny, nx, 2  # grid shape
        y, x = flow.arange(ny, device=d, dtype=t), flow.arange(nx, device=d, dtype=t)

        yv, xv = flow.meshgrid(y, x, indexing="ij")
        grid = flow.stack((xv, yv), 2).expand(shape) - 0.5  # add grid offset, i.e. y = 2.0 * x - 0.5
        anchor_grid = (self.anchors[i] * self.stride[i]).view((1, self.na, 1, 1, 2)).expand(shape)
        return grid, anchor_grid
